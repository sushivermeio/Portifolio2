   private void buttonCalcularActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_buttonCalcularActionPerformed


       //busca o capital digitado e converte pata float


       //ao invés de colocar 'vírgula', utilize '.'


       String capitalDigitado = inputCapital.getText();


       float capitalFloat = converteFloat(capitalDigitado);


      


       //busca o tempo digitado e converte integer


       String tempoDigitado = inputTempo.getText();


       int tempoInt = converteInt(tempoDigitado);


      


       //busca a taxa digitada e converte para float


       String taxaDigitado = inputJuros.getText();


       float taxaFloat = converteFloat(taxaDigitado);


      


       if(capitalFloat==0.00f || taxaFloat==0.00f || tempoInt==0){


           JOptionPane.showMessageDialog(null,"Verifique os dados inseridos\n São aceitos apenas números");


           inputCapital.setText("");


           inputTempo.setText("");


           inputJuros.setText("");


           inputCapital.requestFocus();


       }else{


           float taxaFloatConvertida = taxaFloat / 100;







           //cálculo de Juros Simples


           //J=c.i.t


           float jurosResultado = capitalFloat * tempoInt * taxaFloatConvertida;







           //cálculo do montante







           float montanteResultado = jurosResultado + capitalFloat;







           //colocando juros e montante em duas casas decimais


           DecimalFormat df = new DecimalFormat("0.00");


           String jurosFormatado = df.format(jurosResultado);


           String montanteFormatado = df.format(montanteResultado);







           JOptionPane.showMessageDialog(null, "O Juro resultante é de R$"+jurosFormatado+"\n"+"O montante é R$"+montanteFormatado);


      


       }


      


   }//GEN-LAST:event_buttonCalcularActionPerformed

   private float converteFloat(String inputUsuario){


       try {


           float valorConvertido = Float.parseFloat(inputUsuario);


           return valorConvertido;


       }catch(Exception erro){


           return 0.00f;


       }


   }


  


    private int converteInt(String inputUsuario){


       try {


           int valorConvertido = Integer.parseInt(inputUsuario);


           return valorConvertido;


       }catch(Exception erro){


           return 0;


       }


   }
